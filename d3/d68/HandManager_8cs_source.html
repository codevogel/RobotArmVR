129 {
130 Animator handAnimator = leftRight.Equals(HandType.LEFT) ? HandAnimatorL : HandAnimatorR;
131
132 ResetTriggers(handAnimator);
133
134 HandPose currentPose = GetCurrentPose(leftRight);
135 bool canIdle = currentPose.Equals(fromPose);
136 bool isIdle = currentPose.Equals(HandPose.IDLE);
137 bool hasSwitched = false;
138
139 XRCustomController currentController = leftRight.Equals(HandType.LEFT) ? LeftController : RightController;
140
141 switch (toPose)
142 {
143 case HandPose.IDLE:
144 if (canIdle)
145 {
146 currentController.GetComponentInChildren<CapsuleCollider>().enabled = false;
147 handAnimator.SetTrigger("Idle");
148 hasSwitched = true;
149 }
150 break;
151 case HandPose.JOYSTICK_GRAB:
152 if (leftRight.Equals(HandType.RIGHT))
153 {
154 handAnimator.SetTrigger("JoystickGrab");
155 hasSwitched = true;
156 }
157 break;
158 case HandPose.GRAB:
159 if (isIdle && GetHeldObject(leftRight) != null)
160 {
161 handAnimator.SetTrigger("Grab");
162 hasSwitched = true;
163 }
164 break;
165 case HandPose.POINT:
166 if (isIdle)
167 {
168 currentController.GetComponentInChildren<CapsuleCollider>().enabled = true;
169 handAnimator.SetTrigger("Point");
170 hasSwitched = true;
171 }
172 break;
173 default:
174 throw new ArgumentException();
175 }
176 if (hasSwitched)
177 {
178 SetCurrentPose(leftRight, toPose);
179 }
