110 public float ChangeAxis(int axis, Transform axisTransform)
111 {
112 float axisValue = 0;
113
114 //Grab the specific rotational value from the transform
115 switch (axis)
116 {
117 case 0:
118 axisValue = axisTransform.localRotation.eulerAngles.y;
119 break;
120
121 case 1:
122 case 2:
123 case 4:
124 axisValue = axisTransform.localRotation.eulerAngles.z;
125 break;
126
127 case 3:
128 case 5:
129 axisValue = axisTransform.rotation.x;
130 break;
131 }
132
133 //Calculate if the value should move over to positive or negative
134 switch (axis)
135 {
136 case 0:
137 case 1:
138 case 2:
139 case 4:
140 if (axisValue >= 355 && axisValue <= 360
141 && axisValues[axis].axisRotation >= 0 && axisValues[axis].axisRotation <= 5 && axisValues[axis].negative)
142 {
143 axisValues[axis].negative = false;
144 }
145
146 if (axisValue >= 0 && axisValue <= 5
147 && axisValues[axis].axisRotation >= 355 && axisValues[axis].axisRotation <= 360 && !axisValues[axis].negative)
148 {
149 axisValues[axis].negative = true;
150 }
151 break;
152
153 case 3:
154 case 5:
155 if (axisValue>=0.8f &&axisValue<=1 &&axisValues[axis].axisRotation>=-1&&axisValues[axis].axisRotation<=0.8f&&!axisValues[axis].negative)
156 {
157 axisValues[axis].negative = false;
158 }
159
160 if (axisValues[axis].axisRotation >= -1 && axisValue <= 0.8f && axisValues[axis].axisRotation >= 0.8f&& axisValues[axis].negative)
161 {
162 axisValues[axis].negative = true;
163 }
164 break;
165 }
166
167 return axisValue;

296 private void ChangePositionDisplay()
297 {
298 positionParent.GetChild(2).GetComponent<TextMeshProUGUI>().text = null;

306 //Update each individual axis in axis mode and show if its positive or negative
307 for (int x = 0; x < axisValues.Length - 1; x++)
308 {
309 switch (x)
310 {
311 case 0:
312 case 2:
313 case 4:
314 //Display the value in negative
315 if (axisValues[x].negative)
316 {
317 message += "-"+Math.Round(axisValues[x].axisRotation, 2) + "\n";
318 break;
319 }
320 message += Math.Round(maxAxisRange - axisValues[x].axisRotation, 2) + "\n";
321 break;
322 case 1:
323 if (axisValues[x].negative)
324 {
325 message += Math.Round(axisValues[x].axisRotation, 2) + "\n";
326 break;
327 }
328 message += "-" + Math.Round(maxAxisRange - axisValues[x].axisRotation, 2) + "\n";
329 break;
330 case 3:
331 case 5:
332 float rotation = (axisValues[x].axisRotation / 2) + 0.5f;
333 float rotationDisplay= Mathf.Lerp(-maxAxisRange, maxAxisRange, rotation);
334 message += Math.Round(rotationDisplay, 2) + "\n";

344 for (int x = 0; x < axisValues.Length; x++)

346 switch (x)

348 case 0:

352 break;

356 break;
357 case 2:
358 float percentileZ = CalculatePercentile(minPosition.z, maxPosition.z, flangeTransform.position.y);
359 message += Math.Round(Mathf.Lerp(minRangeZ, maxRangeZ, percentileZ), 2) + " mm\n";
360 break;
361 }
362 }
