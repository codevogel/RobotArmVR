169 public void GenerateRound()
170 {
171#if UNITY_ASSERTIONS
172 Assert.IsNotNull(_goalPrefab, $"{this.name} does not have a goal prefab set.");
173 Assert.IsNotNull(_obstaclePrefab, $"{this.name} does not have an obstacle prefab set.");
174 Assert.IsNotNull(_pointer, $"{this.name} does not have a pointer set and can't register when a goal or obstacle has been touched.");
175 Assert.IsTrue(_radius > 0, $"{this.name} has an invalid radius. The spawn radius must be higher than 0.");
176 Assert.IsTrue(_deadzone >= 0, $"{this.name} has an invalid deadzone. The deadzone radius must be at least 0.");
177 Assert.IsTrue(_radius > _deadzone, $"{this.name} has an invalid deadzone. The deadzone must be smaller than the spawn radius.");
178#endif
179
180 ClearSpawnedObjects();
181
182 int toSpawn = _obstaclesPerRound + 1;
183 float radius = _goalPrefab.GetComponent<Collider>().bounds.extents.x * 1.2f;
184
185 for (int i = 0; i < toSpawn; i++)
186 {
187 Vector3 pos;
188 bool isInvalid;
189 int iterations = default;
190 do
191 {
192 isInvalid = false;
193
194 if (iterations++ > 100)
195 {
196 Debug.LogWarning("Exceeded max iteration count for a randomized position.");
197
198 // if the goal position has not been created keep on forcing it
199 if (i == 0)
200 {
201 continue;
202 }
203
204 break;
205 }
206
207 pos = GenerateRandomPoint();
208 Vector3 worldPos = pos + transform.position;
209
210 // prevent spawning inside of the flange
211 if (IsOverlappingSphere(worldPos, radius, _pointer.transform.position, _pointer.bounds.extents.x * 1.2f))
212 {
213 isInvalid = true;
214 continue;
215 }
216
217 // check previously generated points to prevent overlap
218 int childrenCount = _authoredChildren.Count;
219 for (int j = 0; j < childrenCount; j++)
220 {
221 if (IsOverlappingSphere(worldPos, radius, _authoredChildren[j].transform.position, radius))
222 {
223 isInvalid = true;
224 break;
225 }
226 }
227
228 // if the state is invalid, stop early and iterate again
229 if (isInvalid)
230 continue;
231
232
233 // spawn a goal for the first index, obstacles for anything after
234 if (i == 0)
235 {
236 var goal = Instantiate(_goalPrefab, this.transform);
237 goal.transform.localPosition = pos;
238 goal.Filter = _pointer;
239 goal.OnTriggerEnterRelay.AddListener(() =>
240 {
241 Score++;
242 OnGoalTouched?.Invoke();
243
244 if (_minigameCoroutine == null)
245 _minigameCoroutine = StartCoroutine(MinigameCoroutine());
246
247
248 GenerateRound();
249 });
250
251 _authoredChildren.Add(goal);
252 }
253 else
254 {
255 var obstacle = Instantiate(_obstaclePrefab, this.transform);
256 obstacle.transform.localPosition = pos;
257 obstacle.Filter = _pointer;
258
259 obstacle.OnTriggerEnterRelay.AddListener(() =>
260 {
261 if (Score > 0)
262 {
263 Score--;
264 }
265 OnObstacleTouched?.Invoke();
266
267
268 if (_minigameCoroutine == null)
269 _minigameCoroutine = StartCoroutine(MinigameCoroutine());
270
271 GenerateRound();
272 });
273
274 _authoredChildren.Add(obstacle);
275 }
276 } while (isInvalid);
277 }
278 }

298 private Vector3 GenerateRandomPoint()
299 {
300 Vector3 randomPoint = Random.insideUnitCircle;
301 var deadzoneOffset = randomPoint.normalized * _deadzone;
302
303 randomPoint *= Random.Range(_deadzone, _radius - _deadzone);
304
305 randomPoint += deadzoneOffset;
306
307 randomPoint.z = randomPoint.y;
308 randomPoint.y = (Random.value * _verticalRadius) + _verticalOffset;
309
310 return randomPoint;
311 }

321 private static bool IsOverlappingSphere(Vector3 posA, float radiusA, Vector3 posB, float radiusB)
322 {
323 Vector3 difference = posA - posB;
324 return difference.sqrMagnitude < (radiusA * radiusA) + (radiusB * radiusB);
