397 private void HandleInput(out bool move, Vector2 joystickInput)
398 {
399 if (joystickInteractor.joystickPressed)
400 {
401 // Check whether rotation is more than threshold
402 if (Math.Abs(joystickInteractor.TiltAngle) > joystickInteractor.TiltThreshold)
403 {
404 move = true;
405 selectedArticulator = axisSetOne ? 2 : 5;
406 rotationDirection = joystickInteractor.TiltAngle > 0 ? RotationDirection.Positive : RotationDirection.Negative;
407 return;
408 }
409 }
410 else if (joystickInput.magnitude > joystickThreshold)
411 {
412 bool modifyingX = Math.Abs(joystickInput.x) > Math.Abs(joystickInput.y) ? true : false;
413
414 if (modifyingX)
415 {
416 move = true;
417 selectedArticulator = axisSetOne ? 0 : 3;
418 rotationDirection = joystickInput.x > 0 ? RotationDirection.Negative : RotationDirection.Positive;
419 return;
420 }
421 else
422 {
423
424 move = true;
425 selectedArticulator = axisSetOne ? 1 : 4;
426 if (selectedArticulator == 1)
427 {
428 rotationDirection = joystickInput.y > 0 ? RotationDirection.Negative : RotationDirection.Positive;
429 }
430 else
431 {
432 rotationDirection = joystickInput.y > 0 ? RotationDirection.Positive : RotationDirection.Negative;
433 }
434 return;
435 }
436 }
437 move = false;
438 }
