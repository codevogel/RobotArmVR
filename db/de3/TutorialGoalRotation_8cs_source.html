165 private void Update()
166 {
167 var currentStep = _steps[_currentStepIndex];
168
169 Quaternion targetRotation = currentStep.Axis switch
170 {
171 Axis.X => Quaternion.Euler(currentStep.TargetRotation, 0, 0),
172 Axis.Y => Quaternion.Euler(0, currentStep.TargetRotation, 0),
173 Axis.Z => Quaternion.Euler(0, 0, currentStep.TargetRotation),
174 _ => throw new System.NotImplementedException()
175 };
176
177 // counter rotate the highlight to make it appear as if it is not rotating
178 currentStep.Highlight.localRotation = Quaternion.Inverse(currentStep.Observing.localRotation) * targetRotation;
179
180 // don't check for the angle if it should not be captured at this point in time
181 if (!_capture)
182 return;
183
184 var angle = Quaternion.Angle(currentStep.Observing.localRotation, targetRotation);
185
186 if (currentStep.AllowSymetrical)
187 {
188 angle = Mathf.Min(angle, 180 - angle);
189 }
190
191 // if the observed axis rotation is close enough to the step's goal rotation, start advancing to the next step.
192 if (angle <= _tollerance)
193 {
194 if (_advanceCoroutine == null)
195 {
196 _advanceCoroutine = StartCoroutine(Advance());
